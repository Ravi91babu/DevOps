    CREATE FUNCTION dbo.ConvertCurrency 
    (
        @rate DECIMAL(10,4), 
        @amount DECIMAL(10,4)
    )
    RETURNS DECIMAL(10,4)
    AS
    BEGIN
        DECLARE @Result DECIMAL(10,4);
     
        SET @Result = (SELECT @amount / @rate);
     
        RETURN @Result;
    END;
